\usepackage{xparse}

\DeclareDocumentCommand \subsubsectiontext{o mm} {%
	\IfNoValueTF {#1}
	{\subsubsection[#2]{#2\hfill\faded (#3)}}
	{\subsubsection[#1]{#2\hfill\faded (#3)}}
}
\DeclareDocumentCommand \paragraphtext{o mm} {%
	\IfNoValueTF {#1}
	{\paragraph[#2]{#2\hfill\faded (#3)}}
	{\paragraph[#1]{#2\hfill\faded (#3)}}
}
\DeclareDocumentCommand \subparagraphtext{o mm} {%
	\IfNoValueTF {#1}
	{\subparagraph[#2]{#2\hfill\faded (#3)}}
	{\subparagraph[#1]{#2\hfill\faded (#3)}}
}

\DeclareDocumentCommand\subsubsectionoccurances{omm}
	{\subsubsectiontext[#1]{#2}{#3 occurances}}
\DeclareDocumentCommand\paragraphoccurances{omm}
	{\paragraphtext[#1]{#2}{#3 occurances}}
\DeclareDocumentCommand\subparagraphoccurances{omm}
	{\subparagraphtext[#1]{#2}{#3 occurances}}
